--Задача 1
SELECT *
FROM Водители
WHERE Место прописки = 'еее' AND Стаж NOT IN (3, 5, 6, 10)

--Задача 2
SELECT вод.№ паспорта
FROM Водители as вод
LEFT JOIN Диспетчерская as дс
ON вод.Код водителя = дс.Код водителя
WHERE дс.Путевка IS NULL

--Задача 3
SELECT ав.Регистрационный № авто, дс.Время прибытия, дс.Время отбытия, мш.Пункт назначения
FROM Автомобили as ав, Диспетчерская as дс, Маршрут as мш
LEFT JOIN Диспетчерская as дс ON ав.Код авто = дс.Код авто
LEFT JOIN Маршрут as мш ON дс.Путевка = мш.Код маршрута
WHERE ав.Год выпуска авто = 2015
